import { Injectable } from '@angular/core';
import { Player, Position } from '../models/player.model';

@Injectable({ providedIn: 'root' })
export class TeamService {

  generateTeams(presentPlayers: Player[], shuffle: boolean = false) {
    let team1: Player[] = [];
    let team2: Player[] = [];
    const positionOrder: Position[] = ['Gardien', 'D√©fenseur', 'Milieu', 'Attaquant'];
    let pickTeam1First = true;

    positionOrder.forEach(pos => {
      let playersInPos = presentPlayers.filter(p => p.position === pos);

      // --- LOGIQUE DE VARIATION ---
      if (shuffle) {
        // On m√©lange al√©atoirement les joueurs de cette position
        playersInPos = playersInPos.sort(() => Math.random() - 0.5);
        // Puis on les trie par score, mais les joueurs ayant la m√™me note 
        // ou une note proche seront dans un ordre diff√©rent √† chaque fois.
        playersInPos.sort((a, b) => b.score - a.score);
      } else {
        // Tirage standard : strictement par score
        playersInPos.sort((a, b) => b.score - a.score);
      }

      playersInPos.forEach((player) => {
        if (pickTeam1First) team1.push(player);
        else team2.push(player);
        pickTeam1First = !pickTeam1First;
      });
    });

    return this.ensureNumericBalance(team1, team2);
  }

  private ensureNumericBalance(t1: Player[], t2: Player[]) {
    while (Math.abs(t1.length - t2.length) > 1) {
      if (t1.length > t2.length) {
        const player = t1.pop();
        if (player) t2.push(player);
      } else {
        const player = t2.pop();
        if (player) t1.push(player);
      }
    }
    return { team1: t1, team2: t2 };
  }
}
